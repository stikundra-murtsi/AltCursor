< AltCursor маркировка и расстановка курсоров >

# Описание

- def 	- базовый курсор.
- quest 	- вопрос.
- prg1	- выполнение.
- prg2	- жесткое выполнение.
- cross	- крестик.
- goriz	- горизонтальное изменение размера.
- vert	- вертикальное изменение размера.
- move	- перемещение, ну и добавление.
- text	- курсор редактирования текста.
- err	   - курсор запрета.
- h11	   - рука разжата.
- h12	   - рука сжата.
- h2	   - палец.
- cell 	- курсор управления таблицами.
- pen 	- карандаш.
- zoom*  - zoom-in приближение, zoom-out отдаление.
- diag1  - диагональное изменение размера.
- diag2  - диагональное изменение размера.

d2 - d1
|     |
d1 - d2

# Расстановка

- def    - left_ptr, default, arrow, alias
- quest  - question_arrow, context-menu, help
- prg    - left_ptr_watch, watch, wait, progress 
- cross  - cross, crosshair, tcross
- goriz  - sb_h_double_arrow, left_side, right_side, w-resize, e-resize, row-resize, ew-resize
- vert   - sb_v_double_arrow, top_side, bottom_side, n-resize, ns-resize, s-resize, col-resize
- move   - move, plus, dnd_move, fleur, all-resize, all-scroll
- text   - xterm, vertical-text
- err    - crossed_circle, dnd_no_drop, not-allowed
- h11    - hand1
- h12    - grabbing, grab
- h2     - pointer, hand2
- diag1  - fd_double_arrow, top_right_corner, bottom_left_corner, nwse-resize, ne-resize, sw-resize
- diag2  - bd_double_arrow, top_left_corner, bottom_right_corner, nesw-resize, se-resize
- cell   - cell
- zoom*  - zoom-in zoom-out
- pen	   - pencil
